#include <stdio.h>
#include <limits.h>

int matrixChain(int p[], int n) {
    int m[n][n];
    for(int i=1;i<n;i++) m[i][i]=0;
    for(int L=2;L<n;L++) {
        for(int i=1;i<n-L+1;i++) {
            int j=i+L-1;
            m[i][j]=INT_MAX;
            for(int k=i;k<j;k++) {
                int q=m[i][k]+m[k+1][j]+p[i-1]*p[k]*p[j];
                if(q<m[i][j]) m[i][j]=q;
            }
        }
    }
    return m[1][n-1];
}

int main() {
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++) scanf("%d",&arr[i]);
    printf("Minimum number of multiplications is %d\\n", matrixChain(arr,n));
    return 0;
}
