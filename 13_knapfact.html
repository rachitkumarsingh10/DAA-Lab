Fractional Knapsack Problem Implementation

C

#include <stdio.h>
#include <stdlib.h>

typedef struct { int val; int wt; } Item;

int cmp(const void *a,const void *b){
    double r1=(double)((Item*)a)->val/((Item*)a)->wt;
    double r2=(double)((Item*)b)->val/((Item*)b)->wt;
    return (r1<r2)?1:-1;
}

double fractionalKnapsack(Item arr[], int n, int W){
    qsort(arr,n,sizeof(Item),cmp);
    double total=0;
    for(int i=0;i<n;i++){
        if(arr[i].wt <= W){ total+=arr[i].val; W-=arr[i].wt; }
        else { total+=arr[i].val*((double)W/arr[i].wt); break; }
    }
    return total;
}

int main(){
    int n,W; scanf("%d %d",&n,&W);
    Item arr[n];
    for(int i=0;i<n;i++) scanf("%d %d",&arr[i].val,&arr[i].wt);
    printf("Maximum value in Fractional Knapsack: %.2lf\\n", fractionalKnapsack(arr,n,W));
    return 0;
}
