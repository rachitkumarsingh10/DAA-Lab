Naive String Matching Implementation

C

#include <stdio.h>
#include <string.h>

void naiveSearch(char txt[], char pat[]) {
    int n=strlen(txt), m=strlen(pat);
    for(int i=0;i<n-m+1;i++){
        int j;
        for(j=0;j<m;j++) if(txt[i+j]!=pat[j]) break;
        if(j==m) printf("Pattern found at index %d\\n",i);
    }
}

int main(){
    char txt[100], pat[100];
    scanf("%s %s", txt, pat);
    naiveSearch(txt, pat);
    return 0;
}
