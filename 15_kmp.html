#include <stdio.h>
#include <string.h>

void computeLPS(char pat[], int m, int lps[]){
    int len=0,i=1;
    lps[0]=0;
    while(i<m){
        if(pat[i]==pat[len]) lps[i++]=++len;
        else if(len!=0) len=lps[len-1];
        else lps[i++]=0;
    }
}

void KMPSearch(char txt[], char pat[]){
    int n=strlen(txt), m=strlen(pat);
    int lps[m];
    computeLPS(pat,m,lps);
    int i=0,j=0;
    while(i<n){
        if(txt[i]==pat[j]) { i++; j++; }
        if(j==m) { printf("Pattern found at index %d\\n", i-j); j=lps[j-1]; }
        else if(i<n && txt[i]!=pat[j]){
            if(j!=0) j=lps[j-1];
            else i++;
        }
    }
}

int main(){
    char txt[100], pat[100];
    scanf("%s %s", txt, pat);
    KMPSearch(txt, pat);
    return 0;
}
